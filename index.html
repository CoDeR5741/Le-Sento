    <script>

        gsap.registerPlugin(ScrollTrigger);

        // --- 1. إعداد النجوم (Starfield) ---

        const starContainer = document.getElementById('starfield');

        const starCount = 200;

        for (let i = 0; i < starCount; i++) {

            let star = document.createElement('div');

            star.className = 'star';

           

            // توزيع عشوائي

            let x = (Math.random() - 0.5) * window.innerWidth;

            let y = (Math.random() - 0.5) * window.innerHeight;

            let z = Math.random() * 1000; // عمق عشوائي

           

            star.style.width = Math.random() * 3 + 'px';

            star.style.height = star.style.width;

           

            // نستخدم GSAP لتعيين الخصائص المبدئية

            gsap.set(star, { x: x, y: y, z: z, opacity: Math.random() });

            starContainer.appendChild(star);

        }

        // --- 2. الجدول الزمني الرئيسي (Master Timeline) ---

        const tl = gsap.timeline({

            scrollTrigger: {

                trigger: ".scroll-height",

                start: "top top",

                end: "bottom bottom",

                scrub: 1.5 // زيادة الرقم تجعل الحركة أكثر نعومة وثقلاً

            }

        });

        // --- المرحلة أ: التقريب عبر النجوم والنص ---

        tl.to(".star", {

            z: "+=2000", // النجوم تتحرك نحوك بسرعة

            opacity: 0,

            duration: 5,

            ease: "power1.in"

        }, 0);

        tl.to("#heroText", {

            scale: 15, // تكبير النص حتى يغطي الشاشة

            z: 500,

            opacity: 0,

            filter: "blur(10px)",

            duration: 3,

            ease: "power2.inOut"

        }, 0);

        // --- المرحلة ب: اختراق البوابة (The Portal) ---

        // الصورة تبدأ صغيرة وتكبر جداً لدرجة أننا نمر "من خلالها"

        tl.fromTo("#portal",

            { scale: 0.5, z: -100, opacity: 0.5 },

            { scale: 50, z: 1000, opacity: 1, duration: 4, ease: "power3.in" },

            0.5 // تبدأ بعد النص بقليل

        );

       

        // إخفاء البوابة في نهاية حركتها

        tl.to("#portal", { opacity: 0, duration: 0.5 }, 4);

        // --- المرحلة ج: ظهور المنتج (Landing) ---

        // الخلفية البيضاء تأتي من العمق لتغطي الشاشة

        tl.to("#product", {

            opacity: 1,

            z: 0, // يعود للمكان الطبيعي

            duration: 2,

            ease: "power2.out"

        }, 3.5);

        // --- المرحلة د: تحويل السكتش إلى حقيقة ---

        // هذا يحدث فقط عندما يستقر المنتج

            ease: "none"

        }, 5); // في نهاية التمرير
